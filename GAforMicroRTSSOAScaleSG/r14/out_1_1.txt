for(Unit u){
	if(u.OpponentHasUnitInPlayerRange()) then {
		u.idle()
	} else {
		if(HasLessNumberOfUnits(Heavy,25)) then {
			if(HasNumberOfWorkersHarvesting(4)) then {
				u.moveToUnit(Ally,Weakest)
			} else {
				u.train(Worker,Left)
				u.harvest()
			}
		}
	}
	u.moveToUnit(Enemy,MostHealthy)
}
