		u.attack(LessHealthy)
		u.harvest()
		u.moveToUnit(Ally,LessHealthy)
	} else {
		if(HaveQtdUnitsAttacking(2)) then {
			u.attack(Strongest)
		} else {
			u.harvest()
		}
		u.build(Ranged,Right)
		u.idle()
	}
	if(u.OpponentHasUnitThatKillsUnitInOneAttack()) then {
		if(OpponentHasNumberOfUnits(Ranged,0)) then {
			e
		} else {
			e
		}
	} else {
		u.train(Worker,Right)
	}

		u.idle()
	} else {

	}
